## FIX #1: XSS Prevention in Broadcast Messages

### File: app.py (send_broadcast route)
# BEFORE:
message = request.form.get('message', '').strip()

# AFTER:
import html
message = html.escape(request.form.get('message', '').strip())

## FIX #4: Tiebreaker Validation

### File: app.py (submit_answers route)
# CHANGE:
tiebreaker = int(request.form.get('tiebreaker', 0) or 0)

# TO:
try:
    tiebreaker = int(request.form.get('tiebreaker', 0) or 0)
    if tiebreaker < 0 or tiebreaker > 100:
        flash('⚠️ Tiebreaker must be between 0 and 100', 'error')
        return redirect(url_for('team_play'))
except ValueError:
    tiebreaker = 0
